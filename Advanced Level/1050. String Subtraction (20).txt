#include<iostream>
#include<string>
using namespace std;
int main()
{
	bool hash[256]={0};//¿‡À∆hash±Ì£ø
	string str1;

	getline(cin,str1);
	//getline(cin,str2);
	char ch;
	while((ch=getchar())!='\n')

		hash[ch]=1;
	}

		if(!hash[str1[i]])

	}
